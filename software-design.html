// SecurityUtils.kt - Advanced cryptographic security
object SecurityUtils {
    fun hashPasswordWithSalt(password: String, salt: ByteArray? = null): Pair<String, ByteArray> {
        val actualSalt = salt ?: generateSalt()
        val digest = MessageDigest.getInstance("SHA-256")
        
        digest.update(actualSalt)
        val hashedBytes = digest.digest(password.toByteArray(Charsets.UTF_8))
        val hashedPassword = hashedBytes.joinToString("") { "%02x".format(it) }
        
        return Pair(hashedPassword, actualSalt)
    }
    
    fun verifyPassword(password: String, storedHash: String, salt: ByteArray?): Boolean {
        val (hashedPassword, _) = if (salt != null && salt.isNotEmpty()) {
            hashPasswordWithSalt(password, salt)
        } else {
            Pair(hashPasswordSimple(password), ByteArray(0))
        }
        return constantTimeEquals(hashedPassword, storedHash)
    }
    
    private fun constantTimeEquals(a: String, b: String): Boolean {
        if (a.length != b.length) return false
        var result = 0
        for (i in a.indices) {
            result = result or (a[i].code xor b[i].code)
        }
        return result == 0
    }
}
// ValidationUtils.kt - Comprehensive input validation
object ValidationUtils {
    data class ValidationResult(val isValid: Boolean, val errorMessage: String = "")
    
    fun validateLoginInput(username: String, password: String): ValidationResult {
        if (username.isBlank() || password.isBlank()) {
            return ValidationResult(false, Constants.EMPTY_FIELDS_ERROR)
        }
        
        if (username.trim().length < Constants.MIN_USERNAME_LENGTH) {
            return ValidationResult(false, Constants.USERNAME_TOO_SHORT_ERROR)
        }
        
        if (password.length < Constants.MIN_PASSWORD_LENGTH) {
            return ValidationResult(false, Constants.PASSWORD_TOO_SHORT_ERROR)
        }
        
        if (!isValidUsername(username.trim())) {
            return ValidationResult(false, "Username can only contain letters, numbers, and underscores")
        }
        
        return ValidationResult(true)
    }
    
    fun isValidPhoneNumber(phoneNumber: String): Boolean {
        val digitsOnly = phoneNumber.replace(Regex("[^0-9]"), "")
        if (digitsOnly.length < Constants.PHONE_NUMBER_MIN_LENGTH) return false
        
        val phoneRegex = Regex("^[+]?[0-9]{10,15}$")
        return phoneRegex.matches(digitsOnly)
    }
}
// NotificationHelper.kt - Enterprise notification management
object NotificationHelper {
    fun sendLowStockNotification(context: Context, itemName: String, quantity: Int) {
        try {
            // Input validation and security checks
            if (itemName.isBlank() || quantity < 0) return
            
            // Android 13+ permission check
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
                if (ContextCompat.checkSelfPermission(context, 
                    android.Manifest.permission.POST_NOTIFICATIONS) != PackageManager.PERMISSION_GRANTED) {
                    return
                }
            }
            
            val sanitizedItemName = itemName.trim().take(50)
            val notification = NotificationCompat.Builder(context, Constants.NOTIFICATION_CHANNEL_ID)
                .setSmallIcon(android.R.drawable.ic_dialog_alert)
                .setContentTitle(Constants.NOTIFICATION_TITLE)
                .setContentText(String.format(Constants.NOTIFICATION_TEXT_TEMPLATE, 
                    sanitizedItemName, quantity))
                .setStyle(NotificationCompat.BigTextStyle()
                    .bigText(String.format(Constants.NOTIFICATION_BIG_TEXT_TEMPLATE, 
                        sanitizedItemName, quantity)))
                .setPriority(NotificationCompat.PRIORITY_HIGH)
                .setCategory(NotificationCompat.CATEGORY_ALARM)
                .setAutoCancel(true)
                .build()
                
            val notificationId = sanitizedItemName.hashCode()
            NotificationManagerCompat.from(context).notify(notificationId, notification)
            
        } catch (e: SecurityException) {
            Log.e(TAG, "Security exception sending notification: ${e.message}")
        }
    }
}
