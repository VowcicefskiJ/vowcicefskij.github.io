// Example: Advanced Security Implementation
fun verifyPassword(password: String, storedHash: String, salt: ByteArray?): Boolean {
    val (hashedPassword, _) = if (salt != null && salt.isNotEmpty()) {
        hashPasswordWithSalt(password, salt)
    } else {
        Pair(hashPasswordSimple(password), ByteArray(0))
    }
    
    // Constant-time comparison prevents timing attacks
    return constantTimeEquals(hashedPassword, storedHash)
}

private fun constantTimeEquals(a: String, b: String): Boolean {
    if (a.length != b.length) return false
    var result = 0
    for (i in a.indices) {
        result = result or (a[i].code xor b[i].code)
    }
    return result == 0
}
